script> // ✅ GET ORDER ID FROM URL const orderId = new URLSearchParams(window.location.search).get("id"); // ✅ UPDATE STATUS FUNCTION function updateStatus() { fetch(`http://127.0.0.1:8000/orders/${orderId}/status`) .then(res => res.json()) .then(data => highlight(data.status)); } // ✅ HIGHLIGHT STEPS function highlight(status) { const steps = [ "PLACED", "ACCEPTED", "PREPARING", "READY", "PICKED_UP", "DELIVERED" ]; steps.forEach(s => { const el = document.getElementById(s); if (el) el.classList.remove("active"); }); // ✅ Activate all steps till current let activate = true; steps.forEach(s => { if (activate) { document.getElementById(s).classList.add("active"); } if (s === status) activate = false; }); } // ✅ AUTO REFRESH (LIVE EFFECT) setInterval(updateStatus, 3000); updateStatus();
